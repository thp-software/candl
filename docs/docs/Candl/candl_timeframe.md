# Candl TimeFrame

```ts
export enum CandlTimeFrame {
  // Candle of 1 minute
  Time1Minute = 0,
  // Candle of 5 minutes
  Time5Minutes = 1,
  // Candle of 15 minutes
  Time15Minutes = 2,
  // Candle of 30 minutes
  Time30Minutes = 3,
  // Candle of 1 hour
  Time1Hour = 4,
  // Candle of 4 hours
  Time4Hours = 5,
  // Candle of 8 hours
  Time8Hours = 6,
  // Candle of 12 hours
  Time12Hours = 7,
  // Candle of 1 day
  Time1Day = 8,
  // Candle of 1 week
  Time1Week = 9,
  // Candle of 1 month
  Time1Month = 10,
  // Candle of 1 year
  Time1Year = 11,
}
```
